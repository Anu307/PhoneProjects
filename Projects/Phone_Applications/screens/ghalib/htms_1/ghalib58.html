[taskiin=satisfaction; iztiraab=anxiety; tapish=burn; majaal=strength]

[teG=sword, aab=sharpen]

[jumbish=motion, bosaa=kiss]

pyaalaa gar nahii.n detaa na de, sharaab to de

[oak=palm of the hand]
