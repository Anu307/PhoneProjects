[gardish=bad time/luck; mudaam=always]

[harf=alphabet; mukarrar=again]

[uqoobat=pain]

[laal=ruby; zumuruud=emerald; zar=gold; gauhar=pearl]

[dareG=concealed, mahr=sun, maah=moon]
